    **_splitlines helper (B-1 fix) — mirror Python `str.splitlines()` semantics for trailing-newline parity:**

    Python `"a\nb\nc\n".splitlines()` returns `['a','b','c']` (3 elements, NO trailing empty). JS `"a\nb\nc\n".split(/\r?\n/)` returns `['a','b','c','']` (4 elements WITH trailing empty). On a file ending with `\n`, `.slice(-200)` from a JS split shifts the window by 1 — the signal Python catches at position N-200+1 is missed, and the signal Python would NOT catch may be incorrectly included.

    Implement a helper that pops the trailing empty string after split:

    ```
    function _splitlines(s) {
      const lines = s.split(/\r?\n/);
      if (lines.length > 0 && lines[lines.length - 1] === '') lines.pop();
      return lines;
    }
    ```

    This helper mirrors Python `splitlines()` for the common `\n` / `\r\n` cases used by Claude Code transcripts. (Python `splitlines()` also handles `\r`-only, `\v`, `\f` etc., but transcripts in practice never contain those — documented as known minor divergence per D-04.)

    **detectSignal function -- port of detect_signal() lines 9-60:**

    ```
    function detectSignal(transcriptPath) {
      if (!transcriptPath) return '';

      let content;
      try {
        content = fs.readFileSync(transcriptPath, 'utf-8');
      } catch (e) {
        // Mirror Python (FileNotFoundError, IOError, PermissionError, UnicodeDecodeError) -> ''
        return '';
      }

      // Last 200 lines window — mirror Python line 50: '\n'.join(content.splitlines()[-200:])
      // CRITICAL: must use _splitlines (which mirrors Python splitlines) — NOT a naive split(/\r?\n/).
      // See B-1 fix in 28-01-PLAN.md: trailing-newline files break window alignment otherwise.
      const lines = _splitlines(content);
      const recent = lines.slice(-200).join('\n');

      if (GSD_PLAN_SIGNALS.some(sig => recent.includes(sig))) return 'gsd-plan-complete';
      if (IMPLEMENTATION_SIGNALS.some(sig => recent.includes(sig))) return 'superpowers-implementation-complete';
      if (REVIEW_SIGNALS.some(sig => recent.includes(sig))) return 'superpowers-review-complete';
      if (HOOKIFY_SIGNALS.some(sig => recent.includes(sig))) return 'hookify-complete';
      return '';
    }
    ```
