Phase resolution bash:
  if [ -n "$ARGUMENTS" ]; then
    PHASE_NUM="$ARGUMENTS"
  else
    PHASE_NUM=$(grep -E '^Phase: [0-9]+' .planning/STATE.md | head -1 | awk '{print $2}')
  fi
  if [ -z "$PHASE_NUM" ]; then
    echo "Could not resolve current phase. Pass phase number explicitly: /super-gsd:sg-{name} <phase>"
    exit 1
  fi
